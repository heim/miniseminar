  class lucid32 {
    package { "apache2":
      ensure => present,
    }

    service { "apache2":
      ensure => running,
      require => Package["apache2"],
    }
    
    file {"indexfile":
           path    => "/var/www/index.html",
           ensure  => present,
           content => "Hei verden",
           require => Package["apache2"],
     }
  }

  include lucid32
